### Array#join

```ruby
s        = "A cat in the hat"
words    = s.split(" ")
combined = words.join(":") # => "A:cat:in:the:hat"
```

```rust
  let s              = ~"A cat in the hat";
  let words: ~[&str] = s.words().collect();
  let combined       = words.connect(":");
```
